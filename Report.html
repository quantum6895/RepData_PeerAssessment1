Using average interval for imputing missing data

data_imputed <- data
for (i in 1:17568) {
  if (is.na(data_imputed$steps[i]))  {
    interval <- data_imputed$interval[i]
    data_imputed$steps[i] <- avg_steps$Steps[avg_steps$interval == interval]
  }
}
